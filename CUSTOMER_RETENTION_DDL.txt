-- Create table CUSTOMERS
CREATE TABLE CUSTOMERS (
    CUSTOMER_ID      CHAR(5)       ,
    CUST_ACCOUNT_NUMBER    CHAR(10),
    CUST_LAST_PURCHASE   DATE,
    CUST_CUSTOMER_HISTORY_YEARS    NUMERIC,
    CUST_AVG_ACCOUNTS_RECEIVABLE NUMERIC,
    CUST_EARLY_PAYMENT_FREQUENCY NUMERIC,
    CUST_SATISFACTION_RATING NUMERIC,
    CONSTRAINT CUSTOMERS_PK PRIMARY KEY (CUSTOMER_ID) );
 
-- Create table EMPLOYEES
CREATE TABLE EMPLOYEES (
    EMPLOYEE_ID      CHAR(4)       ,
    EMP_EMPLOYEE_HISTORY_YEARS    NUMERIC,
    EMP_PAST_PERFORMANCE_METRIC   NUMERIC,
    EMP_NEW_CUSTOMER_ACQUISITION_PAST_YEAR   NUMERIC,
    CONSTRAINT EMPLOYEES_PK PRIMARY KEY (EMPLOYEE_ID) );


-- Create table RETENTION
CREATE TABLE RETENTION (
    ASSIGNMENT_ID     VARCHAR2(3)     ,
    CUSTOMER_ID      CHAR(5)     ,
    EMPLOYEE_ID      CHAR(4)     ,
    RETENTION_TARGET INTEGER,
    CONSTRAINT RETENTION_PK PRIMARY KEY (ASSIGNMENT_ID, CUSTOMER_ID, EMPLOYEE_ID),
    CONSTRAINT RETENTION_CUSTOMER_ID_FK FOREIGN KEY (CUSTOMER_ID)
        REFERENCES CUSTOMERS,    
    CONSTRAINT RETENTION_EMPLOYEE_ID_FK FOREIGN KEY (EMPLOYEE_ID) 
        REFERENCES EMPLOYEES  );
